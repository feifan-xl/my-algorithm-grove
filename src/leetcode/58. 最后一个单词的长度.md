### 题解

```js
/**
 * @param {string} s
 * @return {number}
 */
var lengthOfLastWord = function(s) {
    let len = s.length, start = s[len - 1] != ' ', res = 0;
    for (let i = len - 1; i >= 0; i--) {
        
        const cur = s[i]
        if (cur !== ' ') {
            start = true
            res++
        } else {
            if (start) {
                return res;
            }
        }
    }
    return res;
};
```
