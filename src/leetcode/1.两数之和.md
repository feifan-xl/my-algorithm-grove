## 两数之和

[链接](https://leetcode.cn/problems/two-sum/)

### 题解

```js
var twoSum = function(nums, target) {
    var map = {};
    for (let i = 0 ; i < nums.length; i++) {
        if (map[nums[i]] !== undefined) {
            return [map[nums[i]], i];
        }
        map[target - nums[i]] = i
    }
    return []
};
```
